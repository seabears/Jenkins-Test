```
# Jenkins 공식 이미지 기반
FROM jenkins/jenkins:lts-jdk11

# root 권한으로 패키지 설치
USER root

# 패키지 업데이트 후 gcc, cppcheck 설치
RUN apt-get update && \
    apt-get install -y gcc cppcheck && \
    apt-get clean && \
    rm -rf /var/lib/apt/lists/*

# 다시 jenkins 유저로 변경
USER jenkins
```
